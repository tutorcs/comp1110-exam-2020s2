### Q6.3A  [5 Marks]  Complexity

````
static <T extends Comparable<T>> void f(List<T> in) {
    boolean s;
    do {
        s = false;
        for (var i = 1; i < in.size() - 1; i += 2) {
            if (in.get(i).compareTo(in.get(i + 1)) > 0) {
                var temp = in.get(i);
                in.set(i, in.get(i + 1));
                in.set(i + 1, temp);
                s = true;
            }
        }
        for (var i = 0; i < in.size() - 1; i += 2) {
            if (in.get(i).compareTo(in.get(i + 1)) > 0) {
                var temp = in.get(i);
                in.set(i, in.get(i + 1));

                s = true;
            }
        }
    } while (s);
}
````
